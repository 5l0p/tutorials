DSP - Part 14: Sample Playback
2026-09-07


Every sampler, drum machine, and audio player starts the same way: read PCM samples from
a file, store them in memory, then replay them at a pitch and speed that may be
completely different from the original. The interesting problem is fractional positioning
— playing back at a rate that does not land on integer sample indices. Get the
interpolation wrong and you hear aliasing, muffled high frequencies, or faint chirping.
Get it right and a 44.1 kHz snare sounds clean at any playback rate.


What You Will Learn


How to parse a RIFF/WAV header in C and extract raw PCM samples.
How to store audio data as a normalised float buffer.
How to implement one-shot and looping playback with a fractional phase accumulator.
How linear interpolation works and where it falls short.
How cubic Hermite interpolation improves quality for pitched playback.
How a complete playback voice struct ties all of this together.


The RIFF/WAV Format


A WAV file is a RIFF container. The outer RIFF chunk wraps a WAVE form type, followed by
one or more sub-chunks. For basic PCM audio we care about exactly two sub-chunks:
fmt  (format parameters) and data (the raw PCM samples). All multi-byte
integers in RIFF are little-endian.



/* Minimal WAV header layout (all values little-endian) */

/* RIFF chunk */
char  riff_id[4];        /* "RIFF" */
uint32_t riff_size;      /* file size - 8 */
char  wave_id[4];        /* "WAVE" */

/* fmt  sub-chunk */
char  fmt_id[4];         /* "fmt " (note trailing space) */
uint32_t fmt_size;       /* 16 for PCM */
uint16_t audio_format;   /* 1 = PCM, 3 = IEEE float */
uint16_t num_channels;   /* 1 = mono, 2 = stereo */
uint32_t sample_rate;    /* e.g. 44100 or 48000 */
uint32_t byte_rate;      /* sample_rate * num_channels * bits/8 */
uint16_t block_align;    /* num_channels * bits/8 */
uint16_t bits_per_sample;/* 8, 16, 24, or 32 */

/* data sub-chunk */
char  data_id[4];        /* "data" */
uint32_t data_size;      /* number of bytes of PCM data */
/* PCM samples follow immediately */



Real WAV files may have extra chunks (LIST, smpl, cue, etc.) between fmt and data. A
robust parser must skip unknown chunks by reading their size field and seeking forward.



#include <stdio.h>
#include <stdint.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    float    *samples;       /* normalised float, mono (interleaved if stereo) */
    int       num_samples;   /* total sample frames */
    int       num_channels;
    int       sample_rate;
} WavBuffer;

static uint16_t read_u16le(FILE *f) {
    uint8_t b[2];
    fread(b, 1, 2, f);
    return (uint16_t)(b[0] | (b[1] << 8));
}

static uint32_t read_u32le(FILE *f) {
    uint8_t b[4];
    fread(b, 1, 4, f);
    return (uint32_t)(b[0] | (b[1]<<8) | (b[2]<<16) | (b[3]<<24));
}

/* Load a 16-bit PCM WAV file into a WavBuffer.
   Returns 0 on success, non-zero on error. */
int wav_load(const char *path, WavBuffer *wb) {
    FILE *f = fopen(path, "rb");
    if (!f) return -1;

    /* RIFF header */
    char id[4];
    fread(id, 1, 4, f);
    if (memcmp(id, "RIFF", 4) != 0) { fclose(f); return -2; }
    read_u32le(f);           /* riff_size, ignore */
    fread(id, 1, 4, f);
    if (memcmp(id, "WAVE", 4) != 0) { fclose(f); return -3; }

    uint16_t audio_format = 0, num_channels = 0, bits = 0;
    uint32_t sample_rate = 0, data_size = 0;
    int found_fmt = 0, found_data = 0;

    while (!found_data) {
        if (fread(id, 1, 4, f) != 4) break;
        uint32_t chunk_size = read_u32le(f);

        if (memcmp(id, "fmt ", 4) == 0) {
            audio_format  = read_u16le(f);
            num_channels  = read_u16le(f);
            sample_rate   = read_u32le(f);
            read_u32le(f);              /* byte_rate */
            read_u16le(f);              /* block_align */
            bits          = read_u16le(f);
            /* Skip any extra fmt bytes */
            if (chunk_size > 16)
                fseek(f, (long)(chunk_size - 16), SEEK_CUR);
            found_fmt = 1;
        } else if (memcmp(id, "data", 4) == 0) {
            data_size = chunk_size;
            found_data = 1;
        } else {
            /* Unknown chunk: skip */
            fseek(f, (long)chunk_size, SEEK_CUR);
        }
    }

    if (!found_fmt || !found_data || audio_format != 1 || bits != 16) {
        fclose(f);
        return -4;   /* Only 16-bit PCM supported here */
    }

    int total_frames = (int)(data_size / (num_channels * 2));
    float *buf = malloc(total_frames * num_channels * sizeof(float));
    if (!buf) { fclose(f); return -5; }

    /* Read 16-bit signed samples and normalise to [-1, 1] */
    for (int i = 0; i < total_frames * num_channels; i++) {
        int16_t s;
        fread(&s, 2, 1, f);
        buf[i] = (float)s / 32768.0f;
    }

    fclose(f);

    wb->samples      = buf;
    wb->num_samples  = total_frames;
    wb->num_channels = num_channels;
    wb->sample_rate  = (int)sample_rate;
    return 0;
}

void wav_free(WavBuffer *wb) {
    free(wb->samples);
    wb->samples = NULL;
}


The Playback Voice


A playback voice tracks position within the buffer as a floating-point phase. Each
audio sample, the phase advances by rate: a rate of 1.0 plays at the
original pitch, 2.0 plays an octave up, 0.5 an octave down.



typedef enum { PLAYBACK_ONE_SHOT, PLAYBACK_LOOP } PlaybackMode;

typedef struct {
    const WavBuffer *buf;     /* source sample (not owned) */
    double   phase;           /* current read position (fractional index) */
    double   rate;            /* samples-per-output-sample advance */
    float    gain;            /* linear gain [0..1] */
    PlaybackMode mode;
    int      loop_start;      /* sample frame index for loop start */
    int      loop_end;        /* sample frame index for loop end (exclusive) */
    int      active;          /* 0 = silent / finished */
} PlaybackVoice;

void voice_init(PlaybackVoice *v, const WavBuffer *buf,
                double rate, float gain, PlaybackMode mode) {
    v->buf        = buf;
    v->phase      = 0.0;
    v->rate       = rate;
    v->gain       = gain;
    v->mode       = mode;
    v->loop_start = 0;
    v->loop_end   = buf->num_samples;
    v->active     = 1;
}


Linear Interpolation


When phase has a fractional part, the output sample lies between two
adjacent input samples. Linear interpolation blends them in proportion to the fractional
distance. It is cheap (one multiply-add) but introduces a gentle low-pass effect at
high frequencies that becomes audible at rates far from 1.0.



/* Read one mono sample from buf at fractional position pos using linear interp */
static float interp_linear(const float *samples, int len, double pos) {
    int i0 = (int)pos;
    int i1 = i0 + 1;
    if (i1 >= len) i1 = len - 1;   /* clamp at end */
    float frac = (float)(pos - (double)i0);
    return samples[i0] + frac * (samples[i1] - samples[i0]);
}


Cubic Hermite Interpolation


Cubic Hermite interpolation uses four sample points — one before and two after the
fractional position — to fit a smooth cubic polynomial. It preserves slopes at the
sample points, eliminates most of the linear low-pass blur, and still requires no
trigonometric functions. The cost is three extra multiplies compared to linear.



/* Cubic Hermite interpolation across four points p0..p3.
   t is the fractional position in [0, 1] between p1 and p2. */
static float hermite4(float p0, float p1, float p2, float p3, float t) {
    float c0 =  p1;
    float c1 = (p2 - p0) * 0.5f;
    float c2 =  p0 - 2.5f*p1 + 2.0f*p2 - 0.5f*p3;
    float c3 = -0.5f*p0 + 1.5f*p1 - 1.5f*p2 + 0.5f*p3;
    return ((c3*t + c2)*t + c1)*t + c0;
}

static float interp_cubic(const float *samples, int len, double pos) {
    int i1 = (int)pos;
    int i0 = i1 - 1;  if (i0 < 0)    i0 = 0;
    int i2 = i1 + 1;  if (i2 >= len) i2 = len - 1;
    int i3 = i1 + 2;  if (i3 >= len) i3 = len - 1;
    float t = (float)(pos - (double)i1);
    return hermite4(samples[i0], samples[i1], samples[i2], samples[i3], t);
}



Gotcha: at playback rates greater than 1.0 (pitch up) the voice advances more than one
sample per output sample. Cubic interpolation still works correctly because it always
reads the four nearest input samples, regardless of the advance rate. However, for very
high ratios (4x and above) you will start to hear aliasing from the spectral folding of
the original content. Proper anti-aliasing for large-ratio pitch shifts requires a
more expensive resampling filter — out of scope here, but worth knowing about.


Rendering a Playback Voice


/* Render 'frames' mono output samples from voice v into out[].
   Uses cubic Hermite interpolation. */
void voice_render(PlaybackVoice *v, float *out, int frames) {
    if (!v->active) {
        for (int i = 0; i < frames; i++) out[i] = 0.0f;
        return;
    }

    const float *src = v->buf->samples;
    int len = v->buf->num_samples;

    for (int i = 0; i < frames; i++) {
        /* Bounds check */
        if (v->phase >= (double)(len - 1)) {
            if (v->mode == PLAYBACK_LOOP) {
                /* Wrap back to loop_start */
                double loop_len = (double)(v->loop_end - v->loop_start);
                while (v->phase >= (double)v->loop_end)
                    v->phase -= loop_len;
            } else {
                /* One-shot: done */
                out[i] = 0.0f;
                v->active = 0;
                for (int j = i + 1; j < frames; j++) out[j] = 0.0f;
                return;
            }
        }

        out[i] = v->gain * interp_cubic(src, len, v->phase);
        v->phase += v->rate;
    }
}


Pitch Shifting via Playback Rate


To play a sample at a different pitch, compute the rate from the desired frequency ratio.
For MIDI note-based pitch shifting, the rate relative to the root note is:



/* Compute playback rate for a given MIDI note relative to the root note.
   root_note is the MIDI note at which rate = 1.0 (original pitch).
   target_note is the desired MIDI note to play. */
double rate_for_note(int root_note, int target_note) {
    return pow(2.0, (target_note - root_note) / 12.0);
}

/* Example: sample recorded at A4 (MIDI 69), play at C5 (MIDI 72) */
/* rate = pow(2, 3/12.0) = 1.1892...  (about 1.19x speed) */



If the sample's original pitch is not A4, pass the correct root_note. Some WAV files
embed their root note in a smpl chunk; for simple use the root is usually
documented with the sample library or set to 60 (middle C).


Stereo Playback


For stereo WAV files, the samples array is interleaved: L[0], R[0], L[1], R[1], ...
Adjust the read position accordingly:



/* Read one stereo frame at fractional frame position pos.
   Fills *left and *right. */
void interp_stereo(const WavBuffer *wb, double pos, float *left, float *right) {
    int len  = wb->num_samples;
    int i1   = (int)pos;
    float t  = (float)(pos - (double)i1);

    /* Helper: get sample at frame index, channel ch (0=L, 1=R) */
    #define S(idx, ch) wb->samples[((idx) < len ? (idx) : len-1) * 2 + (ch)]

    int i0 = (i1 > 0)      ? i1-1 : 0;
    int i2 = (i1+1 < len)  ? i1+1 : len-1;
    int i3 = (i1+2 < len)  ? i1+2 : len-1;

    *left  = hermite4(S(i0,0), S(i1,0), S(i2,0), S(i3,0), t);
    *right = hermite4(S(i0,1), S(i1,1), S(i2,1), S(i3,1), t);
    #undef S
}


Summary


Loading a WAV file is a matter of parsing its RIFF sub-chunks and converting integer
PCM to float. Playback is a phase accumulator that advances by rate each
sample; linear interpolation is fast but blurry at high pitch ratios; cubic Hermite
is the standard quality/cost trade-off. Looping wraps the phase back to
loop_start when it reaches loop_end. Pitch shifting is
just rate scaling derived from a semitone interval.

interpolation techniques for a wavetable oscillator — the efficient alternative to
computing sin() per sample.
