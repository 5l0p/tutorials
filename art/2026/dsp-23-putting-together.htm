DSP - Part 23: Putting It All Together — a Complete Monophonic Synthesiser

Twenty-two articles of theory, code, and gotchas lead here: a synthesiser you can actually play. It has a wavetable oscillator with band-limited tables, a two-pole lowpass biquad filter with resonance, an ADSR envelope controlling both amplitude and filter cutoff, and it responds to MIDI note-on/off and pitch-bend from any ALSA sequencer source. The audio output goes through SDL2 so the same code builds on Linux, macOS, and Windows. The whole thing fits in 480 lines of C. This article walks the architecture, shows the full source, and closes out the series.

Architecture

The program has three threads:

  1. Main thread: initialises SDL2 audio and ALSA MIDI, then blocks waiting for the user to press Enter (or sends a quit signal). Also handles denormal flags on startup.
  2. MIDI thread: polls the ALSA sequencer for events, parses them with the Part 18 state machine, and writes note/pitch-bend data to the shared voice state through a lightweight spinlock.
  3. Audio thread: the SDL2 callback. Reads voice state, advances the oscillator and envelopes, runs the filter, and writes samples to the output buffer.

The shared voice state is a small struct protected by an SDL mutex. The audio callback holds the lock for less than a microsecond to read parameters; the MIDI thread holds it briefly to write note events. The lock is taken at callback entry and released at exit — short enough that it will never cause underruns at 512-sample buffers.

Shared Voice State

#include <SDL2/SDL.h>
#include <alsa/asoundlib.h>
#include <math.h>
#include <string.h>
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <xmmintrin.h>
#include <pmmintrin.h>

#define SAMPLE_RATE  48000
#define WT_SIZE      2048
#define TWO_PI       (2.0f * 3.14159265358979f)

/* ---- Wavetable (band-limited saw, 8 octave tables) ---- */
#define NUM_WT_TABLES 8
static float wt_tables[NUM_WT_TABLES][WT_SIZE];
static float wt_top_freq[NUM_WT_TABLES];  /* highest freq valid for each table */

/* ---- Biquad filter (Direct Form II transposed) ---- */
typedef struct { float b0,b1,b2,a1,a2,z1,z2; } Biquad;

/* ---- ADSR envelope ---- */
typedef enum { ENV_IDLE, ENV_ATTACK, ENV_DECAY, ENV_SUSTAIN, ENV_RELEASE } EnvState;
typedef struct {
    EnvState state;
    float    env;
    float    att_c, dec_c, sus, rel_c;
} ADSR;

/* ---- Voice parameters (written by MIDI thread, read by audio thread) ---- */
typedef struct {
    int   note_active;     /* 1 = note is held */
    int   note;            /* MIDI note number 0-127 */
    float pitch_bend;      /* semitones, [-2, +2] */
    int   note_off_req;    /* 1 = note-off received, not yet processed */
    SDL_mutex *lock;
} VoiceParams;

/* ---- Synthesiser state (owned by audio thread only) ---- */
typedef struct {
    /* Oscillator */
    double osc_phase;
    double osc_inc;
    int    osc_table;      /* which band-limited table to use */

    /* Filter */
    Biquad filt;
    float  filt_base_cutoff;   /* Hz at full envelope open */
    float  filt_env_amount;    /* semitones of cutoff shift at env peak */

    /* Envelopes */
    ADSR   amp_env;
    ADSR   filt_env;

    /* Note state (local copy, updated from VoiceParams each callback) */
    int   current_note;
    float current_bend;    /* semitones */
    int   gate;            /* 1 = note held */
} SynthVoice;

Building Block Functions

/* Time to one-pole IIR coefficient */
static float tc(float ms) {
    if (ms <= 0.0f) return 0.0f;
    return expf(-1.0f / (ms * 0.001f * SAMPLE_RATE));
}

/* ADSR */
static void adsr_init(ADSR *e, float a, float d, float s, float r) {
    e->att_c = tc(a); e->dec_c = tc(d);
    e->sus   = s;      e->rel_c = tc(r);
    e->env   = 0.0f;   e->state = ENV_IDLE;
}
static void adsr_gate_on(ADSR *e)  { e->state = ENV_ATTACK; }
static void adsr_gate_off(ADSR *e) { if (e->state != ENV_IDLE) e->state = ENV_RELEASE; }

static float adsr_next(ADSR *e) {
    switch (e->state) {
    case ENV_ATTACK:
        e->env = e->att_c * e->env + (1.0f - e->att_c) * 1.05f;
        if (e->env >= 1.0f) { e->env = 1.0f; e->state = ENV_DECAY; }
        break;
    case ENV_DECAY:
        e->env = e->dec_c * e->env + (1.0f - e->dec_c) * e->sus;
        if (fabsf(e->env - e->sus) < 1e-4f) e->state = ENV_SUSTAIN;
        break;
    case ENV_SUSTAIN:
        e->env = e->sus;
        break;
    case ENV_RELEASE:
        e->env = e->rel_c * e->env;
        if (e->env < 1e-5f) { e->env = 0.0f; e->state = ENV_IDLE; }
        break;
    default: e->env = 0.0f; break;
    }
    return e->env;
}

/* Biquad: lowpass, Direct Form II transposed */
static void bq_lowpass(Biquad *f, float fc, float q) {
    float w0  = TWO_PI * fc / SAMPLE_RATE;
    float al  = sinf(w0) / (2.0f * q);
    float cw  = cosf(w0);
    float a0  = 1.0f + al;
    f->b0 = (1.0f - cw) * 0.5f / a0;
    f->b1 = (1.0f - cw)        / a0;
    f->b2 = f->b0;
    f->a1 = -2.0f * cw          / a0;
    f->a2 = (1.0f - al)         / a0;
}
static float bq_process(Biquad *f, float x) {
    float y = f->b0*x + f->z1;
    f->z1   = f->b1*x - f->a1*y + f->z2 + 1e-25f;  /* DC guard */
    f->z2   = f->b2*x - f->a2*y;
    return y;
}

/* Build band-limited saw wavetable set */
static void build_wavetables(void) {
    float nyquist = (float)SAMPLE_RATE * 0.5f;
    float top = 55.0f;
    for (int t = 0; t < NUM_WT_TABLES; t++) {
        wt_top_freq[t] = top * 2.0f;
        int max_h = (int)(nyquist / top);
        if (max_h < 1) max_h = 1;
        memset(wt_tables[t], 0, sizeof(wt_tables[t]));
        for (int h = 1; h <= max_h; h++) {
            float amp = 1.0f / h;
            for (int i = 0; i < WT_SIZE; i++)
                wt_tables[t][i] += amp * sinf(TWO_PI * h * i / WT_SIZE);
        }
        /* Normalise */
        float pk = 0.0f;
        for (int i = 0; i < WT_SIZE; i++) {
            float v = fabsf(wt_tables[t][i]);
            if (v > pk) pk = v;
        }
        if (pk > 0.0f) for (int i = 0; i < WT_SIZE; i++) wt_tables[t][i] /= pk;
        top *= 2.0f;
    }
}

/* Select wavetable for given frequency */
static int wt_table_for(float freq) {
    for (int t = 0; t < NUM_WT_TABLES - 1; t++)
        if (freq <= wt_top_freq[t]) return t;
    return NUM_WT_TABLES - 1;
}

/* Wavetable oscillator: linear interpolation */
static float wt_osc(double *phase, double inc, int table) {
    int   i0 = (int)*phase;
    int   i1 = (i0 + 1) & (WT_SIZE - 1);
    float t  = (float)(*phase - (double)i0);
    float s  = wt_tables[table][i0] + t * (wt_tables[table][i1] - wt_tables[table][i0]);
    *phase  += inc;
    if (*phase >= WT_SIZE) *phase -= WT_SIZE;
    return s;
}

/* MIDI note to frequency */
static float note_to_freq(int note, float bend_semitones) {
    return 440.0f * powf(2.0f, ((float)(note - 69) + bend_semitones) / 12.0f);
}

The Audio Callback

static VoiceParams *g_params;   /* shared with MIDI thread */
static SynthVoice   g_voice;

static void audio_callback(void *userdata, Uint8 *stream, int len) {
    /* Flush-to-zero: silence denormals (x86 only; see Part 13) */
    _MM_SET_FLUSH_ZERO_MODE(_MM_FLUSH_ZERO_ON);
    _MM_SET_DENORMALS_ZERO_MODE(_MM_DENORMALS_ZERO_ON);

    SynthVoice *v = &g_voice;
    float      *buf = (float *)stream;
    int         n   = len / (int)sizeof(float);

    /* Snapshot MIDI state under lock */
    SDL_LockMutex(g_params->lock);
    int   note         = g_params->note;
    float bend         = g_params->pitch_bend;
    int   gate         = g_params->note_active;
    int   note_off_req = g_params->note_off_req;
    g_params->note_off_req = 0;
    SDL_UnlockMutex(g_params->lock);

    /* Note-on: new note */
    if (gate && note != v->current_note) {
        v->current_note = note;
        v->current_bend = bend;
        float freq = note_to_freq(note, bend);
        v->osc_table = wt_table_for(freq);
        v->osc_inc   = (double)WT_SIZE * freq / SAMPLE_RATE;
        adsr_gate_on(&v->amp_env);
        adsr_gate_on(&v->filt_env);
        v->gate = 1;
    }
    /* Note-off */
    if (note_off_req && v->gate) {
        adsr_gate_off(&v->amp_env);
        adsr_gate_off(&v->filt_env);
        v->gate = 0;
    }
    /* Pitch bend (continuous) */
    if (v->gate && fabsf(bend - v->current_bend) > 0.001f) {
        v->current_bend = bend;
        float freq = note_to_freq(v->current_note, bend);
        v->osc_table = wt_table_for(freq);
        v->osc_inc   = (double)WT_SIZE * freq / SAMPLE_RATE;
    }

    for (int i = 0; i < n; i++) {
        float amp_env  = adsr_next(&v->amp_env);
        float filt_env = adsr_next(&v->filt_env);

        /* Filter cutoff modulation: envelope shifts cutoff up to filt_env_amount semitones */
        float cutoff = v->filt_base_cutoff *
                       powf(2.0f, filt_env * v->filt_env_amount / 12.0f);
        if (cutoff > 20000.0f) cutoff = 20000.0f;
        bq_lowpass(&v->filt, cutoff, 1.5f);   /* resonant filter: Q=1.5 */

        float osc_out = wt_osc(&v->osc_phase, v->osc_inc, v->osc_table);
        float filt_out = bq_process(&v->filt, osc_out);
        buf[i] = filt_out * amp_env * 0.4f;   /* 0.4 = safe output level */
    }
}

Gotcha: recomputing the biquad coefficients every sample is expensive. For a slow-moving filter envelope you can update the coefficients once per block instead (outside the inner loop). For the capstone demo the simplicity of per-sample updates outweighs the CPU cost; in a production synthesiser with many voices you would pre-compute the biquad coefficients once per block.

The MIDI Thread

typedef struct {
    snd_seq_t   *seq;
    int          port_id;
    VoiceParams *params;
    int          running;
} MidiThread;

static void *midi_thread_fn(void *arg) {
    MidiThread *mt = (MidiThread *)arg;

    int npfds = snd_seq_poll_descriptors_count(mt->seq, POLLIN);
    struct pollfd *pfds = malloc((size_t)npfds * sizeof(struct pollfd));
    snd_seq_poll_descriptors(mt->seq, pfds, (unsigned int)npfds, POLLIN);

    while (mt->running) {
        if (poll(pfds, (nfds_t)npfds, 100) <= 0) continue;

        snd_seq_event_t *ev;
        while (snd_seq_event_input(mt->seq, &ev) > 0) {
            SDL_LockMutex(mt->params->lock);
            switch (ev->type) {
            case SND_SEQ_EVENT_NOTEON:
                if (ev->data.note.velocity > 0) {
                    mt->params->note        = ev->data.note.note;
                    mt->params->note_active = 1;
                } else {
                    /* Note-on velocity 0 = note-off (running status optimisation) */
                    mt->params->note_off_req = 1;
                    mt->params->note_active  = 0;
                }
                break;
            case SND_SEQ_EVENT_NOTEOFF:
                mt->params->note_off_req = 1;
                mt->params->note_active  = 0;
                break;
            case SND_SEQ_EVENT_PITCHBEND:
                /* ALSA delivers pitch bend -8192..+8191; map to ±2 semitones */
                mt->params->pitch_bend =
                    (float)ev->data.control.value / 8192.0f * 2.0f;
                break;
            default: break;
            }
            SDL_UnlockMutex(mt->params->lock);
        }
    }

    free(pfds);
    return NULL;
}

main(): Wiring It Together

int main(int argc, char **argv) {
    /* Build wavetables */
    build_wavetables();

    /* Shared params */
    VoiceParams params;
    memset(&params, 0, sizeof(params));
    params.lock = SDL_CreateMutex();
    g_params    = &params;

    /* Initialise voice */
    memset(&g_voice, 0, sizeof(g_voice));
    adsr_init(&g_voice.amp_env,   10.0f, 200.0f, 0.7f, 400.0f);
    adsr_init(&g_voice.filt_env,   5.0f, 300.0f, 0.3f, 500.0f);
    g_voice.filt_base_cutoff = 800.0f;
    g_voice.filt_env_amount  = 36.0f;   /* 3 octaves of filter sweep */
    g_voice.current_note     = 60;
    bq_lowpass(&g_voice.filt, g_voice.filt_base_cutoff, 1.5f);

    /* Open ALSA MIDI sequencer */
    snd_seq_t *seq;
    int midi_port;
    if (snd_seq_open(&seq, "default", SND_SEQ_OPEN_INPUT, 0) < 0) {
        fprintf(stderr, "Cannot open ALSA sequencer\n");
        return 1;
    }
    snd_seq_set_client_name(seq, "monosynth");
    midi_port = snd_seq_create_simple_port(seq, "input",
        SND_SEQ_PORT_CAP_WRITE | SND_SEQ_PORT_CAP_SUBS_WRITE,
        SND_SEQ_PORT_TYPE_MIDI_GENERIC | SND_SEQ_PORT_TYPE_APPLICATION);

    /* Auto-connect first available hardware MIDI input if arg given */
    if (argc > 1) {
        int src_c = 0, src_p = 0;
        if (sscanf(argv[1], "%d:%d", &src_c, &src_p) == 2) {
            snd_seq_port_subscribe_t *sub;
            snd_seq_port_subscribe_alloca(&sub);
            snd_seq_addr_t sender = {(unsigned char)src_c, (unsigned char)src_p};
            snd_seq_addr_t dest   = {(unsigned char)snd_seq_client_id(seq),
                                     (unsigned char)midi_port};
            snd_seq_port_subscribe_set_sender(sub, &sender);
            snd_seq_port_subscribe_set_dest(sub, &dest);
            snd_seq_subscribe_port(seq, sub);
            printf("Connected to %d:%d\n", src_c, src_p);
        }
    } else {
        printf("Usage: %s client:port\n  (find ports with: aconnect -l)\n", argv[0]);
    }

    /* Start MIDI thread */
    MidiThread mt = { seq, midi_port, &params, 1 };
    pthread_t midi_tid;
    pthread_create(&midi_tid, NULL, midi_thread_fn, &mt);

    /* Open SDL2 audio */
    if (SDL_Init(SDL_INIT_AUDIO) < 0) {
        fprintf(stderr, "SDL_Init: %s\n", SDL_GetError());
        return 1;
    }
    SDL_AudioSpec desired, obtained;
    memset(&desired, 0, sizeof(desired));
    desired.freq     = SAMPLE_RATE;
    desired.format   = AUDIO_F32SYS;
    desired.channels = 1;
    desired.samples  = 512;
    desired.callback = audio_callback;
    desired.userdata = NULL;
    SDL_AudioDeviceID dev = SDL_OpenAudioDevice(NULL, 0, &desired, &obtained,
        SDL_AUDIO_ALLOW_SAMPLES_CHANGE);
    if (dev == 0) { fprintf(stderr, "SDL_OpenAudioDevice: %s\n", SDL_GetError()); return 1; }
    SDL_PauseAudioDevice(dev, 0);

    printf("Monophonic synth running. Press Enter to quit.\n");
    getchar();

    /* Shutdown */
    mt.running = 0;
    pthread_join(midi_tid, NULL);
    SDL_CloseAudioDevice(dev);
    SDL_Quit();
    snd_seq_close(seq);
    SDL_DestroyMutex(params.lock);
    return 0;
}

What the series did not cover: physical modelling (waveguides, resonators), sample-based synthesis with multi-sample velocity layers, surround audio, MIDI 2.0, hardware interfacing (I2S on embedded targets), or any of the frequency-domain synthesis techniques beyond the phase vocoder sketch in Part 12. Those are threads to pull next.

The code in this series is intentionally minimal — no error recovery, no GUI, no multi-voice polyphony. Each article is a foundation, not a finished product. Take the pieces, read what they do, and build from there.

Summary

The monophonic synthesiser is 480 lines that demonstrate end-to-end real-time audio programming in C. ALSA MIDI provides the note events; SDL2 provides the audio callback; the oscillator, filter, and envelopes fill the buffer. The threading model — MIDI thread writes parameters under a mutex, audio callback reads them — is the pattern used by every real synthesiser, scaled up to more voices, more parameters, and more DSP.
