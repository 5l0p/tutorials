Containers - Part 7: pivot_root & the Container Rootfs
2026-08-05


pivot_root() is the syscall that makes a container's filesystem
isolation real. Unlike chroot(), which merely changes path resolution,
pivot_root() restructures the kernel's mount tree so that the old root
mount becomes inaccessible. Combined with a new mount namespace, there is no
path back to the host filesystem. This article walks through the syscall, the
"pivot_root dance" required to use it, building a minimal rootfs with BusyBox,
and a complete Go runtime implementation.


What You Will Learn

A complete Go implementation using unix.PivotRoot and the re-exec pattern
Why pivot_root + unmount is categorically stronger than chroot


How pivot_root Works


The signature is:



#include <sys/syscall.h>
int pivot_root(const char *new_root, const char *put_old);



pivot_root() has no glibc wrapper in older versions; call it via
syscall(SYS_pivot_root, new_root, put_old). It:



Makes new_root the new root mount of the current mount namespace.
Moves the old root mount to put_old (which must be under new_root).
After the call, the old root is accessible at put_old and can be unmounted.



Constraints enforced by the kernel:



new_root and put_old must be directories.
new_root must be a mount point (not just a directory).
put_old must be under new_root.
The filesystem of new_root must not be the same as the current root.


The pivot_root Dance in C


The standard sequence is:



Bind-mount new_root over itself (to make it a mount point if it is not already).
Create a directory new_root/.old_root to receive the old root.
Call pivot_root(new_root, new_root/.old_root).
chdir("/") to move into the new root.
Unmount /.old_root (the old host root).
Remove the /.old_root directory.



#define _GNU_SOURCE
#include <sched.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/mount.h>
#include <sys/stat.h>
#include <sys/syscall.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <unistd.h>

static inline int pivot_root(const char *new_root, const char *put_old) {
    return syscall(SYS_pivot_root, new_root, put_old);
}

#define STACK_SIZE (1 * 1024 * 1024)
#define ROOTFS     "/tmp/rootfs"

static int container_main(void *arg) {
    (void)arg;

    /* 1. Remount everything private */
    if (mount(NULL, "/", NULL, MS_REC | MS_PRIVATE, NULL) == -1) {
        perror("mount private"); return 1;
    }

    /* 2. Bind-mount new_root over itself to make it a mount point */
    if (mount(ROOTFS, ROOTFS, NULL, MS_BIND | MS_REC, NULL) == -1) {
        perror("bind mount rootfs"); return 1;
    }

    /* 3. Create put_old directory */
    char put_old[256];
    snprintf(put_old, sizeof(put_old), "%s/.old_root", ROOTFS);
    mkdir(put_old, 0700);

    /* 4. pivot_root */
    if (pivot_root(ROOTFS, put_old) == -1) {
        perror("pivot_root"); return 1;
    }

    /* 5. chdir into new root */
    if (chdir("/") == -1) {
        perror("chdir"); return 1;
    }

    /* 6. Unmount old root */
    if (umount2("/.old_root", MNT_DETACH) == -1) {
        perror("umount old root"); return 1;
    }
    rmdir("/.old_root");

    /* Mount /proc for the new PID namespace */
    mkdir("/proc", 0755);
    mount("proc", "/proc", "proc", 0, NULL);

    printf("pivot_root complete - running in new rootfs\n");

    char *argv[] = { "/bin/sh", NULL };
    execv("/bin/sh", argv);
    perror("execv");
    return 1;
}

int main(void) {
    char *stack = malloc(STACK_SIZE);
    if (!stack) { perror("malloc"); return 1; }

    pid_t pid = clone(container_main,
                      stack + STACK_SIZE,
                      CLONE_NEWNS | CLONE_NEWPID | SIGCHLD,
                      NULL);
    if (pid == -1) { perror("clone"); return 1; }

    waitpid(pid, NULL, 0);
    free(stack);
    return 0;
}


Building a Minimal Rootfs with BusyBox


The rootfs needs at least a shell and basic utilities. A statically-linked
BusyBox binary provides them all in one file:



#!/bin/sh
set -e
ROOTFS=/tmp/rootfs

# Create directory structure
mkdir -p $ROOTFS/{bin,sbin,etc,proc,sys,dev,tmp,usr/bin,usr/sbin}

# Download static BusyBox
curl -Lo $ROOTFS/bin/busybox \
  https://busybox.net/downloads/binaries/1.35.0-x86_64-linux-musl/busybox
chmod +x $ROOTFS/bin/busybox

# Install all applets as symlinks
chroot $ROOTFS /bin/busybox --install -s /bin

# Basic /etc/passwd and /etc/group
printf 'root:x:0:0:root:/root:/bin/sh\n' > $ROOTFS/etc/passwd
printf 'root:x:0:\n'                     > $ROOTFS/etc/group

# Create a few device nodes (needed by many programs)
mknod -m 666 $ROOTFS/dev/null  c 1 3
mknod -m 666 $ROOTFS/dev/zero  c 1 5
mknod -m 666 $ROOTFS/dev/urandom c 1 9
mknod -m 622 $ROOTFS/dev/tty   c 5 0

echo "Rootfs built at $ROOTFS"



Run this script as root (mknod requires privileges), then run the C program above
to pivot into it.


Go Implementation


The Go standard library does not have a pivot_root wrapper in
syscall, but golang.org/x/sys/unix does:



package main

import (
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
    "syscall"

    "golang.org/x/sys/unix"
)

const rootfs = "/tmp/rootfs"

func pivotRoot(newRoot string) error {
    putOld := filepath.Join(newRoot, ".old_root")

    // Bind-mount newRoot over itself
    if err := unix.Mount(newRoot, newRoot, "", unix.MS_BIND|unix.MS_REC, ""); err != nil {
        return fmt.Errorf("bind mount: %w", err)
    }

    // Create put_old directory
    if err := os.MkdirAll(putOld, 0700); err != nil {
        return fmt.Errorf("mkdir put_old: %w", err)
    }

    // Pivot
    if err := unix.PivotRoot(newRoot, putOld); err != nil {
        return fmt.Errorf("pivot_root: %w", err)
    }

    // chdir into new root
    if err := os.Chdir("/"); err != nil {
        return fmt.Errorf("chdir: %w", err)
    }

    // Unmount old root
    if err := unix.Unmount("/.old_root", unix.MNT_DETACH); err != nil {
        return fmt.Errorf("unmount old root: %w", err)
    }

    return os.Remove("/.old_root")
}

func child() error {
    // Make mounts private
    if err := unix.Mount("", "/", "", unix.MS_REC|unix.MS_PRIVATE, ""); err != nil {
        return fmt.Errorf("mount private: %w", err)
    }

    if err := pivotRoot(rootfs); err != nil {
        return err
    }

    // Mount proc
    os.MkdirAll("/proc", 0755)
    if err := unix.Mount("proc", "/proc", "proc", 0, ""); err != nil {
        return fmt.Errorf("mount proc: %w", err)
    }

    cmd := exec.Command("/bin/sh")
    cmd.Stdin  = os.Stdin
    cmd.Stdout = os.Stdout
    cmd.Stderr = os.Stderr
    return cmd.Run()
}

func main() {
    // Re-exec ourselves with a flag so the child runs child()
    if len(os.Args) > 1 && os.Args[1] == "child" {
        if err := child(); err != nil {
            fmt.Fprintln(os.Stderr, err)
            os.Exit(1)
        }
        return
    }

    // Parent: launch child in new namespaces
    cmd := exec.Command("/proc/self/exe", "child")
    cmd.Stdin  = os.Stdin
    cmd.Stdout = os.Stdout
    cmd.Stderr = os.Stderr
    cmd.SysProcAttr = &syscall.SysProcAttr{
        Cloneflags: syscall.CLONE_NEWNS | syscall.CLONE_NEWPID,
    }

    if err := cmd.Run(); err != nil {
        fmt.Fprintln(os.Stderr, err)
        os.Exit(1)
    }
}



The re-exec pattern (parent launches /proc/self/exe child) is how runc
and most Go container runtimes handle the namespace setup: the parent creates
the namespaces, the child re-executes itself inside those namespaces and
finishes the rootfs configuration.


Why Not Just chroot?


After pivot_root() + umount2(put_old, MNT_DETACH), the host
filesystem is not present in the container's mount namespace. Even a privileged
process inside the container cannot reach the host's /etc or run host
binaries. With chroot() alone, the host's full mount table is still in the
kernel - the old root is simply not reachable through path resolution unless
the process escapes the jail.


Summary


pivot_root() restructures the kernel's mount tree: the new rootfs becomes
the root of the container's mount namespace and the old root is moved to a
subdirectory where it can be unmounted. After that unmount, the host filesystem
is gone from the container's view — not just hidden by path resolution. The
Go re-exec pattern (parent creates namespaces, child re-executes itself to run
inside them) is the standard approach used by runc and most Go-based runtimes.
