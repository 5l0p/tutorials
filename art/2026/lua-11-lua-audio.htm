Lua - Part 11: Scripting a DSP Pipeline from Lua

The DSP series showed how to implement audio building blocks in C: oscillators, biquad filters, envelope followers, compressors. The Lua series showed how to load Lua, call C from Lua, read config tables, and hot-reload scripts. This article joins the two: a Lua DSL that describes a complete audio node graph, and a C loader that reads it and wires real DSP structs. Change the graph description in a text editor, save the file, and the running audio pipeline reconfigures without a restart. The result is a live-coding-style workflow with the performance of C DSP in the audio thread.

What You Will Learn

  • A clean node-graph DSL in Lua using constructor functions (oscillator{}, filter{}).
  • A C loader that reads the graph table and constructs real DSP objects.
  • How to map Lua parameter tables to C struct fields for each node type.
  • How hot-reload works for a live audio graph: tear down, rebuild, resume.
  • Connection between this series and DSP Parts 3 (oscillators) and 5 (biquad filters).
  • Connection to the libpd series for Pd-based DSP alternatives.

The Node Graph DSL

Rather than writing raw tables with a type field (as in Part 6), we use constructor functions. Each function simply tags its argument table with a type marker and appends it to the graph. This reads more naturally and catches typos in node type names at script load time: 

-- pipeline.lua
-- Constructor functions return tagged tables
local function oscillator(params)
    params._type = "oscillator"
    return params
end

local function filter(params)
    params._type = "filter"
    return params
end

local function gain(params)
    params._type = "gain"
    return params
end

local function output(params)
    params._type = "output"
    return params
end

-- Declare the graph
graph = {
    nodes = {
        osc1 = oscillator {
            freq  = 220.0,
            wave  = "sine",
            amp   = 0.5,
        },
        filt1 = filter {
            type   = "lowpass",
            cutoff = 800.0,
            q      = 1.2,
        },
        gain1 = gain {
            db = -6.0,
        },
        out1  = output { channels = 2 },
    },

    -- Signal chain: osc1 -> filt1 -> gain1 -> out1
    chain = { "osc1", "filt1", "gain1", "out1" },

    -- Global sample rate - read by the C loader
    sample_rate = 48000,
}

The constructor functions are pure Lua - they just set a field and return the table. The C loader reads the graph.nodes table and uses _type to dispatch. The graph.chain sequence defines the processing order.

The C Graph Loader

/* graph_loader.c */
#include <lua.h>
#include <lualib.h>
#include <lauxlib.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include "dsp.h"   /* Oscillator, BiquadFilter, GainNode */

#define MAX_NODES 32

typedef enum { NODE_OSC, NODE_FILTER, NODE_GAIN, NODE_OUTPUT } NodeKind;

typedef struct {
    char      name[64];
    NodeKind  kind;
    union {
        Oscillator   osc;
        BiquadFilter filt;
        GainNode     gain;
    } u;
} AudioNode;

typedef struct {
    AudioNode nodes[MAX_NODES];
    int       count;
    int       chain[MAX_NODES];  /* indices into nodes[], in order */
    int       chain_len;
    int       sample_rate;
} AudioGraph;

static AudioGraph g_graph;

/* build_oscillator: read osc params from Lua table at tbl_idx */
static void build_oscillator(lua_State *L, int tbl_idx,
                              AudioNode *node, int sr) {
    node->kind = NODE_OSC;
    osc_init(&node->u.osc, sr);

    lua_getfield(L, tbl_idx, "freq");
    double freq = lua_isnumber(L, -1) ? lua_tonumber(L, -1) : 440.0;
    lua_pop(L, 1);
    osc_set_freq(&node->u.osc, freq);

    lua_getfield(L, tbl_idx, "amp");
    double amp = lua_isnumber(L, -1) ? lua_tonumber(L, -1) : 1.0;
    lua_pop(L, 1);
    osc_set_amp(&node->u.osc, amp);

    lua_getfield(L, tbl_idx, "wave");
    const char *wave = lua_isstring(L, -1) ? lua_tostring(L, -1) : "sine";
    if      (strcmp(wave, "square") == 0) node->u.osc.wave_type = 1;
    else if (strcmp(wave, "saw")    == 0) node->u.osc.wave_type = 2;
    else if (strcmp(wave, "tri")    == 0) node->u.osc.wave_type = 3;
    else                                  node->u.osc.wave_type = 0;
    lua_pop(L, 1);

    printf("  osc '%s' freq=%.1f wave=%s\n", node->name, freq, wave);
}

static void build_filter(lua_State *L, int tbl_idx,
                          AudioNode *node, int sr) {
    node->kind = NODE_FILTER;
    node->u.filt.sample_rate = sr;

    lua_getfield(L, tbl_idx, "cutoff");
    double fc = lua_isnumber(L, -1) ? lua_tonumber(L, -1) : 1000.0;
    lua_pop(L, 1);

    lua_getfield(L, tbl_idx, "q");
    double q = lua_isnumber(L, -1) ? lua_tonumber(L, -1) : 0.707;
    lua_pop(L, 1);

    lua_getfield(L, tbl_idx, "type");
    const char *ftype = lua_isstring(L, -1) ? lua_tostring(L, -1) : "lowpass";
    if (strcmp(ftype, "highpass") == 0)
        biquad_highpass(&node->u.filt, fc, q);
    else
        biquad_lowpass(&node->u.filt, fc, q);
    lua_pop(L, 1);

    printf("  filter '%s' %s cutoff=%.1f q=%.3f\n",
           node->name, ftype, fc, q);
}

static void build_gain(lua_State *L, int tbl_idx, AudioNode *node) {
    node->kind = NODE_GAIN;
    lua_getfield(L, tbl_idx, "db");
    double db = lua_isnumber(L, -1) ? lua_tonumber(L, -1) : 0.0;
    lua_pop(L, 1);
    node->u.gain.linear_gain = (float)pow(10.0, db / 20.0);
    printf("  gain '%s' %.1f dB (%.4f linear)\n",
           node->name, db, (double)node->u.gain.linear_gain);
}

/* load_graph: read graph{} from L and fill g_graph */
int load_graph(lua_State *L) {
    memset(&g_graph, 0, sizeof(g_graph));

    /* graph table */
    lua_getglobal(L, "graph");
    if (!lua_istable(L, -1)) {
        fprintf(stderr, "load_graph: no 'graph' table\n");
        lua_pop(L, 1); return -1;
    }
    int graph_idx = lua_gettop(L);

    /* sample_rate */
    lua_getfield(L, graph_idx, "sample_rate");
    g_graph.sample_rate = (int)luaL_optinteger(L, -1, 44100);
    lua_pop(L, 1);

    /* nodes sub-table */
    lua_getfield(L, graph_idx, "nodes");
    if (!lua_istable(L, -1)) {
        fprintf(stderr, "load_graph: 'graph.nodes' must be a table\n");
        lua_pop(L, 2); return -1;
    }
    int nodes_idx = lua_gettop(L);

    printf("Loading graph (sr=%d):\n", g_graph.sample_rate);

    /* iterate nodes */
    lua_pushnil(L);
    while (lua_next(L, nodes_idx) != 0 && g_graph.count < MAX_NODES) {
        /* stack: [graph, nodes, key, value] */
        if (lua_type(L, -2) == LUA_TSTRING && lua_istable(L, -1)) {
            const char *name = lua_tostring(L, -2);
            int val_idx = lua_gettop(L);
            AudioNode *node = &g_graph.nodes[g_graph.count++];
            strncpy(node->name, name, sizeof(node->name) - 1);

            lua_getfield(L, val_idx, "_type");
            const char *ntype = lua_isstring(L, -1) ? lua_tostring(L, -1) : "";
            lua_pop(L, 1);

            if      (strcmp(ntype, "oscillator") == 0)
                build_oscillator(L, val_idx, node, g_graph.sample_rate);
            else if (strcmp(ntype, "filter") == 0)
                build_filter(L, val_idx, node, g_graph.sample_rate);
            else if (strcmp(ntype, "gain") == 0)
                build_gain(L, val_idx, node);
            else if (strcmp(ntype, "output") == 0) {
                node->kind = NODE_OUTPUT;
                printf("  output '%s'\n", name);
            } else {
                fprintf(stderr, "  unknown node type '%s' for '%s'\n", ntype, name);
                g_graph.count--;
            }
        }
        lua_pop(L, 1);  /* pop value */
    }
    lua_pop(L, 1);  /* pop nodes */

    /* chain sequence */
    lua_getfield(L, graph_idx, "chain");
    if (lua_istable(L, -1)) {
        lua_Integer n = lua_rawlen(L, -1);
        int chain_tbl = lua_gettop(L);
        for (lua_Integer i = 1; i <= n && g_graph.chain_len < MAX_NODES; i++) {
            lua_rawgeti(L, chain_tbl, i);
            const char *chain_name = lua_isstring(L, -1) ? lua_tostring(L, -1) : NULL;
            lua_pop(L, 1);
            if (!chain_name) continue;
            /* find the node by name */
            for (int j = 0; j < g_graph.count; j++) {
                if (strcmp(g_graph.nodes[j].name, chain_name) == 0) {
                    g_graph.chain[g_graph.chain_len++] = j;
                    break;
                }
            }
        }
    }
    lua_pop(L, 1);   /* pop chain */
    lua_pop(L, 1);   /* pop graph */

    printf("Chain: %d nodes\n", g_graph.chain_len);
    return 0;
}

The Audio Render Loop

The audio callback processes samples by running each node in chain order. The previous node's output is the next node's input: 

/* Called from your audio I/O callback (PortAudio, ALSA, CoreAudio, etc.) */
void render_block(float *output_buffer, int frames) {
    for (int i = 0; i < frames; i++) {
        float sample = 0.0f;

        for (int j = 0; j < g_graph.chain_len; j++) {
            AudioNode *node = &g_graph.nodes[g_graph.chain[j]];
            switch (node->kind) {
            case NODE_OSC:
                sample = osc_process(&node->u.osc);
                break;
            case NODE_FILTER:
                sample = biquad_process(&node->u.filt, sample);
                break;
            case NODE_GAIN:
                sample *= node->u.gain.linear_gain;
                break;
            case NODE_OUTPUT:
                output_buffer[i] = sample;
                break;
            }
        }
    }
}

Hot-Reload: Live Pipeline Reconfiguration

Reloading the pipeline means: finish the current block, swap the graph, resume. In a single-threaded application the reload happens between audio callbacks. In a threaded application, use a flag and double-buffering: 

#include <sys/stat.h>
#include <time.h>
#include <stdatomic.h>

static time_t       g_script_mtime = 0;
static const char  *g_script_path  = "pipeline.lua";
static atomic_int   g_reload_pending = 0;

/* Called from non-audio thread (main loop, file watcher) */
void check_for_reload(lua_State *L) {
    struct stat st;
    if (stat(g_script_path, &st) != 0) return;
    if (st.st_mtime == g_script_mtime) return;
    g_script_mtime = st.st_mtime;

    /* Load and validate the new graph before swapping */
    AudioGraph new_graph;
    memset(&new_graph, 0, sizeof(new_graph));

    if (luaL_loadfile(L, g_script_path) != LUA_OK) {
        fprintf(stderr, "reload: %s\n", lua_tostring(L, -1));
        lua_pop(L, 1); return;
    }
    if (lua_pcall(L, 0, 0, 0) != LUA_OK) {
        fprintf(stderr, "reload: %s\n", lua_tostring(L, -1));
        lua_pop(L, 1); return;
    }

    /* Build new graph into temporary storage (not shown for brevity) */
    /* Then signal audio thread via atomic flag                       */
    g_reload_pending = 1;
    printf("Reload ready\n");
}

/* Called from audio thread between blocks */
void maybe_apply_reload(void) {
    if (!atomic_load(&g_reload_pending)) return;
    /* swap g_graph with new graph (details depend on threading model) */
    atomic_store(&g_reload_pending, 0);
    printf("Graph hot-swapped\n");
}

For a production audio application, protect the swap with a lock-free ring buffer or a double-buffer scheme: the audio thread reads from slot A while the main thread builds slot B, then they atomically swap. Never hold a mutex in the audio callback.

Connection to the DSP Series and libpd

Summary

A Lua node graph DSL built from constructor functions gives audio developers a live, editable description of a signal chain. The C loader reads graph.nodes with lua_next iteration, dispatches on _type, and fills real DSP structs. The audio render loop processes the chain in sequence. Hot-reload is luaL_loadfile + lua_pcall again, with an atomic signal to the audio thread to swap the graph between blocks.
