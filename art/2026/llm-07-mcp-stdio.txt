LLM - Part 7: MCP over stdio
2026-06-16


stdio is the simplest MCP transport: the client spawns the server as a child process and
reads/writes JSON-RPC messages over stdin and stdout. No network, no ports, no auth - the
server is only reachable by its parent. This makes it ideal for local tools that need to
run in the same security context as the agent. This article builds a working stdio MCP
server in Go and connects it to the agent loop from article 04.

How to implement tool handlers and return results or errors

The MCP Server


The mcp-go library provides a server.MCPServer type. You add tools
to it, then run it with a transport. For stdio, the transport is
server.NewStdioServer(mcpServer).



// cmd/server/main.go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "os"

    "github.com/mark3labs/mcp-go/mcp"
    "github.com/mark3labs/mcp-go/server"
)

func main() {
    s := server.NewMCPServer(
        "filesystem-tools",
        "1.0.0",
        server.WithToolCapabilities(false),
    )

    // Register: read_file
    s.AddTool(
        mcp.NewTool("read_file",
            mcp.WithDescription("Read the contents of a file at the given path."),
            mcp.WithString("path",
                mcp.Required(),
                mcp.Description("Path to the file to read"),
            ),
        ),
        handleReadFile,
    )

    // Register: write_file
    s.AddTool(
        mcp.NewTool("write_file",
            mcp.WithDescription("Write content to a file, creating it if it does not exist."),
            mcp.WithString("path",
                mcp.Required(),
                mcp.Description("Path to write to"),
            ),
            mcp.WithString("content",
                mcp.Required(),
                mcp.Description("Content to write"),
            ),
        ),
        handleWriteFile,
    )

    // Register: list_dir
    s.AddTool(
        mcp.NewTool("list_dir",
            mcp.WithDescription("List files and directories at the given path."),
            mcp.WithString("path",
                mcp.Required(),
                mcp.Description("Directory to list"),
            ),
        ),
        handleListDir,
    )

    stdio := server.NewStdioServer(s)
    if err := stdio.Listen(context.Background(), os.Stdin, os.Stdout); err != nil {
        fmt.Fprintln(os.Stderr, "server error:", err)
        os.Exit(1)
    }
}


Tool Handlers


func handleReadFile(ctx context.Context, req mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    path, ok := req.Params.Arguments["path"].(string)
    if !ok || path == "" {
        return mcp.NewToolResultError("path is required"), nil
    }
    data, err := os.ReadFile(path)
    if err != nil {
        return mcp.NewToolResultError(err.Error()), nil
    }
    return mcp.NewToolResultText(string(data)), nil
}

func handleWriteFile(ctx context.Context, req mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    path, _ := req.Params.Arguments["path"].(string)
    content, _ := req.Params.Arguments["content"].(string)
    if path == "" {
        return mcp.NewToolResultError("path is required"), nil
    }
    if err := os.WriteFile(path, []byte(content), 0644); err != nil {
        return mcp.NewToolResultError(err.Error()), nil
    }
    result, _ := json.Marshal(map[string]interface{}{
        "written": len(content), "path": path,
    })
    return mcp.NewToolResultText(string(result)), nil
}

func handleListDir(ctx context.Context, req mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    path, _ := req.Params.Arguments["path"].(string)
    if path == "" {
        path = "."
    }
    entries, err := os.ReadDir(path)
    if err != nil {
        return mcp.NewToolResultError(err.Error()), nil
    }
    type entry struct {
        Name  string `json:"name"`
        IsDir bool   `json:"is_dir"`
    }
    var list []entry
    for _, e := range entries {
        list = append(list, entry{Name: e.Name(), IsDir: e.IsDir()})
    }
    data, _ := json.Marshal(list)
    return mcp.NewToolResultText(string(data)), nil
}



Tool handlers return (*mcp.CallToolResult, error). Use mcp.NewToolResultError
for expected failures (file not found, permission denied) - these are reported to the model
as tool results, not protocol errors. Use a non-nil Go error for unexpected failures like
encoding problems; the library turns these into JSON-RPC errors.


Testing the Server Manually


Build and test the server without an agent:



go build ./cmd/server
echo '{"jsonrpc":"2.0","id":1,"method":"initialize","params":{"protocolVersion":"2024-11-05","capabilities":{},"clientInfo":{"name":"test","version":"1.0"}}}' | ./server



You should see the server's capabilities response on stdout. Send a tools/list
request to see the registered tools, and tools/call to invoke one.


The MCP Client


The agent side uses mcp-go's client package. The client spawns the server
subprocess, performs the initialize handshake, and exposes methods to list tools and call them.



// cmd/agent/main.go
package main

import (
    "context"
    "fmt"
    "os"

    "github.com/mark3labs/mcp-go/client"
    "github.com/mark3labs/mcp-go/mcp"
)

func main() {
    apiKey := os.Getenv("OPENAI_API_KEY")
    if apiKey == "" {
        fmt.Fprintln(os.Stderr, "OPENAI_API_KEY not set")
        os.Exit(1)
    }

    ctx := context.Background()

    // Spawn the server as a subprocess
    mcpClient, err := client.NewStdioMCPClient("./server", nil)
    if err != nil {
        fmt.Fprintln(os.Stderr, "mcp client:", err)
        os.Exit(1)
    }
    defer mcpClient.Close()

    // Initialize
    initResult, err := mcpClient.Initialize(ctx, mcp.InitializeRequest{
        Params: mcp.InitializeRequestParams{
            ProtocolVersion: mcp.LATEST_PROTOCOL_VERSION,
            ClientInfo: mcp.Implementation{
                Name:    "agent",
                Version: "1.0",
            },
        },
    })
    if err != nil {
        fmt.Fprintln(os.Stderr, "initialize:", err)
        os.Exit(1)
    }
    fmt.Printf("Connected to MCP server: %s %s\n",
        initResult.ServerInfo.Name, initResult.ServerInfo.Version)

    // List available tools
    toolsResult, err := mcpClient.ListTools(ctx, mcp.ListToolsRequest{})
    if err != nil {
        fmt.Fprintln(os.Stderr, "list tools:", err)
        os.Exit(1)
    }
    fmt.Printf("Available tools: %d\n", len(toolsResult.Tools))
    for _, t := range toolsResult.Tools {
        fmt.Printf("  - %s: %s\n", t.Name, t.Description)
    }

    // Convert MCP tool definitions to OpenAI Tool format
    openaiTools := mcpToolsToOpenAI(toolsResult.Tools)

    // Build and run the agent with MCP dispatch
    agent := NewMCPAgent(apiKey, "gpt-4o-mini", mcpClient, openaiTools)
    result, err := agent.Run(ctx,
        "List the current directory and tell me what files you see.")
    if err != nil {
        fmt.Fprintln(os.Stderr, "agent:", err)
        os.Exit(1)
    }
    fmt.Println(result)
}


Converting MCP Tool Definitions to OpenAI Format


func mcpToolsToOpenAI(mcpTools []mcp.Tool) []Tool {
    var tools []Tool
    for _, t := range mcpTools {
        tools = append(tools, Tool{
            Type: "function",
            Function: ToolFunction{
                Name:        t.Name,
                Description: t.Description,
                Parameters: ToolParameters{
                    Type:       "object",
                    Properties: convertSchema(t.InputSchema.Properties),
                    Required:   t.InputSchema.Required,
                },
            },
        })
    }
    return tools
}


Dispatching via MCP


When the agent receives a tool call, it forwards it to the MCP server:



func (a *MCPAgent) mcpDispatch(ctx context.Context, name, argsJSON string) (string, error) {
    var args map[string]interface{}
    if err := json.Unmarshal([]byte(argsJSON), &args); err != nil {
        return "", err
    }

    result, err := a.client.CallTool(ctx, mcp.CallToolRequest{
        Params: mcp.CallToolRequestParams{
            Name:      name,
            Arguments: args,
        },
    })
    if err != nil {
        return "", err
    }

    if result.IsError {
        if len(result.Content) > 0 {
            if text, ok := result.Content[0].(mcp.TextContent); ok {
                return "", fmt.Errorf("tool error: %s", text.Text)
            }
        }
        return "", fmt.Errorf("tool returned error")
    }

    if len(result.Content) > 0 {
        if text, ok := result.Content[0].(mcp.TextContent); ok {
            return text.Text, nil
        }
    }
    return "", nil
}


Build and Run


go build ./cmd/server
go build ./cmd/agent
export OPENAI_API_KEY=sk-...
./agent



The agent spawns the server, discovers its tools, sends a task, and uses the tools to
complete it. The server process exits when the agent does.


Summary


A stdio MCP server is a plain Go binary that reads JSON-RPC from stdin and
writes JSON-RPC to stdout. The client spawns it with os/exec,
wraps the pipes in an mcp-go client, calls tools/list to
enumerate tools, and converts the results to the agent's tool format. The
server process is isolated — it cannot be reached by anything other than the
agent that spawned it — making stdio appropriate for trusted local tools
that access the filesystem or run shell commands.
