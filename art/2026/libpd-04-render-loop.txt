libpd - Part 4: The Audio Render Loop
2026-06-01

The render loop is where libpd actually does its work: your audio driver calls a callback, you pass buffers to libpd_process_float, and the Pd engine fills them with computed samples. Getting the block sizes, channel interleaving, and callback structure right is essential for glitch-free audio. This article shows how to integrate libpd with PortAudio, SDL2, and ALSA.

How libpd_process_float Works

Pd processes audio in blocks of exactly 64 samples. libpd_process_float runs the scheduler for ticks such blocks:


int libpd_process_float(int ticks,
                        const float *inBuffer,
                        float       *outBuffer);


Buffer layout: samples are interleaved by channel. For stereo output with 256 samples (ticks=4), outBuffer must be 256 * 2 = 512 floats, laid out as L0 R0 L1 R1 ... L255 R255.

The return value is 0 on success. A non-zero return means the DSP graph is not running (you forgot to send a [; pd dsp 1] message, or initialisation failed).

Enabling the DSP Engine

Pd's DSP engine starts in a stopped state. Enable it before the first audio block:


/* Equivalent to clicking "DSP on" in the Pd GUI */
libpd_start_message(1);
libpd_add_float(1.0f);
libpd_finish_message("pd", "dsp");


To stop DSP (useful when the app is in the background):


libpd_start_message(1);
libpd_add_float(0.0f);
libpd_finish_message("pd", "dsp");


Block-Size Alignment

Your audio driver's block size must be a multiple of 64. Common sizes (128, 256, 512, 1024) all work. If your driver gives you an arbitrary size, you must buffer:


#define PD_BLOCK 64

static float ringbuf[PD_BLOCK * 2]; /* one Pd tick of stereo */
static int   ringpos = 0;

void audio_callback(float *out, int nframes, int nch) {
    int i = 0;
    while (i < nframes) {
        int avail = PD_BLOCK - ringpos;
        int copy  = (nframes - i < avail) ? nframes - i : avail;
        memcpy(out + i * nch, ringbuf + ringpos * nch,
               copy * nch * sizeof(float));
        ringpos += copy;
        i       += copy;
        if (ringpos == PD_BLOCK) {
            libpd_process_float(1, NULL, ringbuf);
            ringpos = 0;
        }
    }
}


In practice, prefer 256 or 512 as your driver block size and avoid the complexity of rebuffering.

PortAudio Integration

PortAudio is the most portable audio I/O library and a natural fit for libpd:


#include <portaudio.h>
#include <z_libpd.h>
#include <string.h>

#define SAMPLE_RATE  44100
#define BLOCK_SIZE   512
#define CHANNELS     2
#define TICKS        (BLOCK_SIZE / 64)

static int pa_callback(const void *inBuf, void *outBuf,
                       unsigned long frames,
                       const PaStreamCallbackTimeInfo *ti,
                       PaStreamCallbackFlags flags,
                       void *userData)
{
    float *out = (float *)outBuf;
    const float *in = (const float *)inBuf;

    /* outBuf is frames * CHANNELS floats; already interleaved */
    libpd_process_float(TICKS, in, out);
    return paContinue;
}

int main(void) {
    libpd_init();
    libpd_init_audio(0, CHANNELS, SAMPLE_RATE);

    /* Enable DSP */
    libpd_start_message(1);
    libpd_add_float(1.0f);
    libpd_finish_message("pd", "dsp");

    void *patch = libpd_openfile("synth.pd", ".");

    Pa_Initialize();
    PaStream *stream;
    Pa_OpenDefaultStream(&stream, 0, CHANNELS, paFloat32,
                         SAMPLE_RATE, BLOCK_SIZE, pa_callback, NULL);
    Pa_StartStream(stream);

    Pa_Sleep(5000); /* play for 5 seconds */

    Pa_StopStream(stream);
    Pa_CloseStream(stream);
    Pa_Terminate();

    libpd_closefile(patch);
    return 0;
}


SDL2 Audio Integration

SDL2's audio callback is similar but uses a different type signature:


#include <SDL2/SDL.h>
#include <z_libpd.h>

#define SAMPLE_RATE 44100
#define BLOCK_SIZE  512
#define CHANNELS    2
#define TICKS       (BLOCK_SIZE / 64)

static void sdl_audio_callback(void *userdata,
                                Uint8 *stream,
                                int len)
{
    float *out = (float *)stream;
    int frames = len / (CHANNELS * sizeof(float));
    (void)frames; /* should equal BLOCK_SIZE */

    libpd_process_float(TICKS, NULL, out);
}

int main(void) {
    libpd_init();
    libpd_init_audio(0, CHANNELS, SAMPLE_RATE);

    libpd_start_message(1);
    libpd_add_float(1.0f);
    libpd_finish_message("pd", "dsp");

    void *patch = libpd_openfile("synth.pd", ".");

    SDL_Init(SDL_INIT_AUDIO);

    SDL_AudioSpec want = {
        .freq     = SAMPLE_RATE,
        .format   = AUDIO_F32SYS,
        .channels = CHANNELS,
        .samples  = BLOCK_SIZE,
        .callback = sdl_audio_callback,
    };

    SDL_AudioSpec got;
    SDL_AudioDeviceID dev = SDL_OpenAudioDevice(NULL, 0, &want, &got, 0);
    SDL_PauseAudioDevice(dev, 0); /* start playback */

    SDL_Delay(5000);

    SDL_CloseAudioDevice(dev);
    SDL_Quit();
    libpd_closefile(patch);
    return 0;
}


ALSA Integration (Linux)

For bare-metal Linux deployments without PortAudio:


#include <alsa/asoundlib.h>
#include <z_libpd.h>
#include <stdlib.h>

#define SAMPLE_RATE 44100
#define CHANNELS    2
#define BLOCK_SIZE  512
#define TICKS       (BLOCK_SIZE / 64)

int main(void) {
    libpd_init();
    libpd_init_audio(0, CHANNELS, SAMPLE_RATE);

    libpd_start_message(1);
    libpd_add_float(1.0f);
    libpd_finish_message("pd", "dsp");

    void *patch = libpd_openfile("synth.pd", ".");

    snd_pcm_t *pcm;
    snd_pcm_open(&pcm, "default", SND_PCM_STREAM_PLAYBACK, 0);
    snd_pcm_set_params(pcm, SND_PCM_FORMAT_FLOAT_LE,
                       SND_PCM_ACCESS_RW_INTERLEAVED,
                       CHANNELS, SAMPLE_RATE, 1, 20000);

    float buf[BLOCK_SIZE * CHANNELS];

    for (;;) {
        libpd_process_float(TICKS, NULL, buf);
        snd_pcm_writei(pcm, buf, BLOCK_SIZE);
    }

    snd_pcm_close(pcm);
    libpd_closefile(patch);
    return 0;
}


Thread Safety in the Render Loop

The render loop runs on the audio thread. Key rules:


Do NOT call libpd_openfile or libpd_closefile from the audio thread.
Do NOT allocate memory in the audio callback.
Message sends (libpd_float, libpd_bang) are safe from the audio thread.
When sending messages from a UI or control thread, use a lock-free ring buffer to queue them, then drain the queue at the top of each audio callback.


Measuring DSP Load

To check if libpd is overrunning its time budget, measure wall-clock time around the process call:


#include <time.h>

struct timespec t0, t1;
clock_gettime(CLOCK_MONOTONIC, &t0);
libpd_process_float(TICKS, inBuf, outBuf);
clock_gettime(CLOCK_MONOTONIC, &t1);

double ms = (t1.tv_sec - t0.tv_sec) * 1000.0
          + (t1.tv_nsec - t0.tv_nsec) / 1e6;

/* Budget: BLOCK_SIZE / SAMPLE_RATE * 1000 ms */
double budget = (double)BLOCK_SIZE / SAMPLE_RATE * 1000.0;
if (ms > budget * 0.8)
    fprintf(stderr, "DSP load high: %.2f / %.2f ms\n", ms, budget);


Summary


Call libpd_process_float(ticks, inBuf, outBuf) from your audio callback where
ticks = blockSize / 64. Buffers are interleaved: all channels for sample 0,
then all channels for sample 1. If your driver delivers block sizes that are not multiples
of 64, maintain an alignment buffer that accumulates input until a full 64-sample tick is
available, then drains output back to the driver. Measure wall time across the call and
log a warning when it exceeds 80% of the available budget.
