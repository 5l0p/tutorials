Pico — Part 10: Complete Synthesiser Voice
2026-06-01


Nine articles of infrastructure and DSP primitives lead here: a synthesiser that works.
Plug in a USB cable, connect a PCM5102 DAC to the I2S pins, open a DAW or a software
synthesiser, route MIDI to "Pico MIDI", and you have a monophonic synthesiser with a
wavetable oscillator, a two-pole resonant lowpass filter, and independent ADSR envelopes
for amplitude and filter cutoff modulation. No operating system, no malloc after startup,
no floating-point in the inner sample loop, and the entire audio engine running on a
four-dollar chip. This article assembles all the pieces, explains how they fit together,
and closes the series.

oscillator.h / oscillator.c


/* oscillator.h */
#ifndef OSCILLATOR_H
#define OSCILLATOR_H
#include <stdint.h>

#define OSC_TABLE_SIZE    2048
#define OSC_PHASE_BITS    22

typedef struct {
    uint32_t phase;
    uint32_t phase_inc;
} Oscillator;

void     osc_init(Oscillator *o);
void     osc_set_freq(Oscillator *o, float freq_hz, float sample_rate);
int16_t  osc_tick(Oscillator *o);

#endif



/* oscillator.c */
#include "oscillator.h"

/* Sine table in flash (.rodata) - generated at build time.
   Values: round(32767 * sin(2*pi*i/2048)) for i = 0..2047. */
static const int16_t sine_table[OSC_TABLE_SIZE] = {
    /* ... 2048 values ... truncated for article length ... */
    0, 101, 201, 301, 402, 502, 603, 703
    /* full table generated by tools/gen_sine_table.py */
};

void osc_init(Oscillator *o) {
    o->phase     = 0;
    o->phase_inc = 0;
}

void osc_set_freq(Oscillator *o, float freq_hz, float sample_rate) {
    o->phase_inc = (uint32_t)(
        (freq_hz / sample_rate) * OSC_TABLE_SIZE * (1u << OSC_PHASE_BITS)
    );
}

int16_t osc_tick(Oscillator *o) {
    uint32_t idx = (o->phase >> OSC_PHASE_BITS) & (OSC_TABLE_SIZE - 1);
    int16_t  s   = sine_table[idx];
    o->phase    += o->phase_inc;
    return s;
}


filter.h / filter.c


We use Direct Form II transposed biquad (one stage = two-pole filter). For a resonant
lowpass, quality factor Q controls the resonance peak. Higher Q values produce a
sharper resonance that is audible as a vowel-like emphasis at the cutoff frequency.



/* filter.h */
#ifndef FILTER_H
#define FILTER_H
#include <stdint.h>

typedef struct {
    float b0, b1, b2;
    float a1, a2;
    float z1, z2;
} Filter;

void  filter_init(Filter *f);
void  filter_set_lowpass(Filter *f, float fc, float q, float sample_rate);
float filter_tick(Filter *f, float x);

#endif



/* filter.c */
#include "filter.h"
#include <math.h>
#include <string.h>

void filter_init(Filter *f) {
    memset(f, 0, sizeof(*f));
    f->b0 = 1.0f;   /* pass-through by default */
}

void filter_set_lowpass(Filter *f, float fc, float q, float sample_rate) {
    /* Guard against degenerate values */
    if (fc <   10.0f) fc =   10.0f;
    if (fc > sample_rate * 0.499f) fc = sample_rate * 0.499f;
    if (q  <   0.5f) q  =   0.5f;
    if (q  >  20.0f) q  =  20.0f;

    float w0    = 2.0f * 3.14159265f * fc / sample_rate;
    float alpha = sinf(w0) / (2.0f * q);
    float cosw0 = cosf(w0);
    float a0_inv = 1.0f / (1.0f + alpha);

    f->b0 = ((1.0f - cosw0) * 0.5f) * a0_inv;
    f->b1 = (1.0f - cosw0)          * a0_inv;
    f->b2 = f->b0;
    f->a1 = (-2.0f * cosw0)         * a0_inv;
    f->a2 = (1.0f - alpha)          * a0_inv;
}

float filter_tick(Filter *f, float x) {
    float y  = f->b0 * x + f->z1;
    f->z1    = f->b1 * x - f->a1 * y + f->z2;
    f->z2    = f->b2 * x - f->a2 * y;
    return y;
}


envelope.h / envelope.c


/* envelope.h */
#ifndef ENVELOPE_H
#define ENVELOPE_H
#include <stdint.h>

typedef enum { ENV_IDLE, ENV_ATTACK, ENV_DECAY, ENV_SUSTAIN, ENV_RELEASE } EnvState;

typedef struct {
    EnvState state;
    int32_t  level;
    int32_t  attack_inc;
    int32_t  decay_inc;
    int32_t  sustain_level;
    int32_t  release_inc;
} Envelope;

void    env_init(Envelope *e);
void    env_set(Envelope *e, float attack_ms, float decay_ms,
                uint8_t sustain_127, float release_ms, float sample_rate);
void    env_note_on(Envelope *e);
void    env_note_off(Envelope *e);
int32_t env_tick(Envelope *e);   /* returns Q15 level [0, 32767] */
bool    env_is_idle(const Envelope *e);

#endif



/* envelope.c */
#include "envelope.h"
#include <string.h>

void env_init(Envelope *e) { memset(e, 0, sizeof(*e)); }

void env_set(Envelope *e, float attack_ms, float decay_ms,
             uint8_t sustain_127, float release_ms, float sample_rate) {
    float as = attack_ms  * 0.001f * sample_rate;
    float ds = decay_ms   * 0.001f * sample_rate;
    float rs = release_ms * 0.001f * sample_rate;
    e->sustain_level = (int32_t)((sustain_127 / 127.0f) * 32767.0f);
    e->attack_inc    = (int32_t)(32767.0f / as); if (e->attack_inc  < 1) e->attack_inc  = 1;
    e->decay_inc     = (int32_t)((32767.0f - e->sustain_level) / ds);
                                                 if (e->decay_inc   < 1) e->decay_inc   = 1;
    e->release_inc   = (int32_t)(e->sustain_level / rs); if (e->release_inc < 1) e->release_inc = 1;
    e->state = ENV_IDLE;
    e->level = 0;
}

void env_note_on(Envelope *e)  { e->state = ENV_ATTACK; }
void env_note_off(Envelope *e) { if (e->state != ENV_IDLE) e->state = ENV_RELEASE; }
bool env_is_idle(const Envelope *e) { return e->state == ENV_IDLE; }

int32_t env_tick(Envelope *e) {
    switch (e->state) {
    case ENV_ATTACK:
        e->level += e->attack_inc;
        if (e->level >= 32767) { e->level = 32767; e->state = ENV_DECAY; }
        break;
    case ENV_DECAY:
        e->level -= e->decay_inc;
        if (e->level <= e->sustain_level) { e->level = e->sustain_level; e->state = ENV_SUSTAIN; }
        break;
    case ENV_SUSTAIN:
        break;
    case ENV_RELEASE:
        e->level -= e->release_inc;
        if (e->level <= 0) { e->level = 0; e->state = ENV_IDLE; }
        break;
    case ENV_IDLE:
        break;
    }
    return e->level;
}


voice.h / voice.c


The voice struct holds an oscillator, a filter, and two envelopes: one for amplitude
(scales the output level) and one for filter cutoff (modulates the filter frequency
to produce the characteristic sweep of a subtractive synthesiser).



/* voice.h */
#ifndef VOICE_H
#define VOICE_H
#include <stdint.h>
#include "oscillator.h"
#include "filter.h"
#include "envelope.h"

#define VOICE_SAMPLE_RATE  44100.0f

typedef struct {
    Oscillator osc;
    Filter     filter;
    Envelope   amp_env;    /* amplitude envelope */
    Envelope   flt_env;    /* filter cutoff envelope */

    /* Filter range: cutoff_base is the base cutoff (set by CC),
       flt_env modulates it by up to flt_env_depth semitones. */
    float cutoff_base;
    float flt_env_depth;   /* in Hz, additive */
    float filter_q;
} Voice;

void    voice_init(Voice *v);
void    voice_note_on(Voice *v, uint8_t note, uint8_t velocity);
void    voice_note_off(Voice *v);
int16_t voice_tick(Voice *v);   /* render one sample, returns int16 */

#endif



/* voice.c */
#include "voice.h"
#include <string.h>

void voice_init(Voice *v) {
    osc_init(&v->osc);
    filter_init(&v->filter);
    env_init(&v->amp_env);
    env_init(&v->flt_env);

    v->cutoff_base   = 800.0f;
    v->flt_env_depth = 4000.0f;
    v->filter_q      = 2.5f;

    /* Default ADSR: organ-like amp, fast-decay filter */
    env_set(&v->amp_env,  10.0f, 100.0f, 100, 300.0f, VOICE_SAMPLE_RATE);
    env_set(&v->flt_env,   5.0f,  80.0f,  40, 200.0f, VOICE_SAMPLE_RATE);

    filter_set_lowpass(&v->filter,
                       v->cutoff_base, v->filter_q, VOICE_SAMPLE_RATE);
}

void voice_note_on(Voice *v, uint8_t note, uint8_t velocity) {
    float freq = 440.0f;
    /* Use precomputed table in production (see Part 7) */
    /* Here we use powf once per note-on - acceptable */
    extern float midi_freq_table[128];
    freq = midi_freq_table[note];

    v->osc.phase = 0;   /* reset phase on note-on */
    osc_set_freq(&v->osc, freq, VOICE_SAMPLE_RATE);
    env_note_on(&v->amp_env);
    env_note_on(&v->flt_env);
    (void)velocity;   /* velocity scaling can be added here */
}

void voice_note_off(Voice *v) {
    env_note_off(&v->amp_env);
    env_note_off(&v->flt_env);
}

int16_t voice_tick(Voice *v) {
    /* 1. Advance envelopes */
    int32_t amp_level = env_tick(&v->amp_env);   /* Q15 */
    int32_t flt_level = env_tick(&v->flt_env);   /* Q15 */

    /* 2. Compute filter cutoff: base + envelope modulation
       flt_level is Q15 [0,32767]; scale to [0, flt_env_depth] Hz */
    float cutoff = v->cutoff_base
                   + ((float)flt_level / 32767.0f) * v->flt_env_depth;

    /* 3. Update filter coefficients only when env is moving.
       In ENV_SUSTAIN the cutoff is constant - skip the coefficient recompute. */
    /* For simplicity we always update here; in production guard with a
       "dirty" flag to save ~30 cycles per buffer when in sustain. */
    filter_set_lowpass(&v->filter, cutoff, v->filter_q, VOICE_SAMPLE_RATE);

    /* 4. Generate oscillator sample */
    int16_t raw = osc_tick(&v->osc);

    /* 5. Apply filter (float input/output) */
    float filtered = filter_tick(&v->filter, (float)raw);

    /* 6. Apply amplitude envelope: scale by Q15 level */
    int32_t out = (int32_t)(filtered * amp_level) >> 15;

    /* 7. Clamp to int16 */
    if (out >  32767) out =  32767;
    if (out < -32768) out = -32768;
    return (int16_t)out;
}



There is one soft-float call per sample in voice_tick:
filter_tick. The biquad uses five float multiplies and five adds - roughly
55 cycles. The envelope and oscillator steps are integer arithmetic: roughly 30 cycles
combined. Total per sample: approximately 90-100 cycles. For 256 samples: 25,600 cycles.
Budget at 44.1 kHz: 725,623 cycles. CPU usage: 3.5%. There is room for four more voices
before the Pico runs out of budget.



Note the filter_set_lowpass call inside voice_tick - this calls
sinf and cosf, each roughly 100-150 cycles, once per sample. That
is expensive: ~300 cycles per sample just for coefficient computation, or 76,800 cycles
per buffer (10.6% CPU). In production, add a dirty flag and only recompute when
cutoff changes by more than a threshold:



/* In voice.c: coefficient update guard */
static float last_cutoff = -1.0f;

if (fabsf(cutoff - last_cutoff) > 1.0f) {   /* 1 Hz threshold */
    filter_set_lowpass(&v->filter, cutoff, v->filter_q, VOICE_SAMPLE_RATE);
    last_cutoff = cutoff;
}



With this guard, coefficients update only when the filter envelope is actively moving
(attack and decay stages), not in sustain or idle. In sustain, the dominant cost drops
back to 3.5% CPU.


main.c: Putting It All Together


/* main.c - complete monophonic synthesiser
   Core 0: USB MIDI + parameter dispatch
   Core 1: audio DMA fill loop at 44.1 kHz                    */

#include <string.h>
#include "pico/stdlib.h"
#include "pico/multicore.h"
#include "hardware/dma.h"
#include "hardware/irq.h"
#include "hardware/pio.h"
#include "hardware/sync.h"
#include "tusb.h"
#include "voice.h"
#include "midi.h"

/* --- Audio constants --- */
#define SAMPLE_RATE   44100
#define BUF_FRAMES    256
#define BUF_SAMPLES   (BUF_FRAMES * 2)   /* stereo interleaved */

/* --- Audio buffers (SRAM) --- */
static int16_t buf_a[BUF_SAMPLES];
static int16_t buf_b[BUF_SAMPLES];
static volatile int16_t *fill_buf;
static volatile bool     audio_ready = false;
static int               audio_dma_chan;

/* --- Synthesiser voice (owned by core 1) --- */
static Voice voice;

/* --- Parameter passing (Parts 7-9) --- */
static spin_lock_t *params_lock;
typedef struct { uint8_t note; uint8_t gate; uint8_t vel; } VoiceCmd;
static volatile VoiceCmd pending_cmd;
static volatile bool     cmd_pending = false;

/* --- MIDI freq table (written once at startup by core 0) --- */
float midi_freq_table[128];

/* ---  DMA IRQ (core 1) --- */
static void audio_dma_irq_handler(void) {
    dma_channel_acknowledge_irq0(audio_dma_chan);
    dma_channel_set_read_addr(audio_dma_chan, (void *)fill_buf, true);
    fill_buf    = (fill_buf == buf_a) ? buf_b : buf_a;
    audio_ready = true;
}

/* --- Core 1 audio fill --- */
static void fill_buffer(int16_t *buf) {
    /* Check for pending voice command from core 0 */
    if (cmd_pending) {
        uint32_t save = spin_lock_blocking(params_lock);
        VoiceCmd cmd = pending_cmd;
        cmd_pending  = false;
        spin_unlock(params_lock, save);
        if (cmd.gate) voice_note_on(&voice, cmd.note, cmd.vel);
        else          voice_note_off(&voice);
    }
    for (int i = 0; i < BUF_FRAMES; i++) {
        int16_t s    = voice_tick(&voice);
        buf[i * 2]   = s;
        buf[i * 2+1] = s;
    }
}

static void core1_entry(void) {
    voice_init(&voice);

    /* Fill both buffers before starting DMA */
    fill_buffer(buf_a);
    fill_buffer(buf_b);
    fill_buf = buf_a;

    /* Set up I2S PIO + DMA (see Part 2 for full init).
       Abbreviated here - replace with your i2s_output_init() call. */

    irq_set_exclusive_handler(DMA_IRQ_0, audio_dma_irq_handler);
    irq_set_enabled(DMA_IRQ_0, true);
    dma_channel_set_irq0_enabled(audio_dma_chan, true);
    dma_channel_set_read_addr(audio_dma_chan, buf_b, true);

    multicore_fifo_push_blocking(1);   /* signal core 0: ready */

    while (true) {
        if (audio_ready) {
            audio_ready = false;
            fill_buffer((int16_t *)fill_buf);
        }
        tight_loop_contents();
    }
}

/* --- Core 0 MIDI handler --- */
static void on_midi_msg(const MidiMsg *msg) {
    if (msg->type == 0x90 && msg->data2 > 0) {
        uint32_t save = spin_lock_blocking(params_lock);
        pending_cmd.note = msg->data1;
        pending_cmd.gate = 1;
        pending_cmd.vel  = msg->data2;
        cmd_pending      = true;
        spin_unlock(params_lock, save);
    } else if (msg->type == 0x80 ||
               (msg->type == 0x90 && msg->data2 == 0)) {
        uint32_t save = spin_lock_blocking(params_lock);
        pending_cmd.gate = 0;
        cmd_pending      = true;
        spin_unlock(params_lock, save);
    }
}

int main(void) {
    /* Init MIDI frequency table */
    for (int i = 0; i < 128; i++)
        midi_freq_table[i] = 440.0f * powf(2.0f, (i - 69) / 12.0f);

    /* Init spinlock */
    params_lock = spin_lock_init(spin_lock_claim_unused(true));

    /* Init TinyUSB */
    board_init();
    tusb_init();

    /* Init UART MIDI (optional, alongside USB MIDI) */
    midi_uart_init();

    /* Launch audio core */
    multicore_launch_core1(core1_entry);
    multicore_fifo_pop_blocking();   /* wait for core 1 ready */

    /* Core 0 main loop */
    while (true) {
        tud_task();
        usb_midi_poll_with_callback(on_midi_msg);
        midi_uart_poll_with_callback(on_midi_msg);
    }
}

Series Retrospective


This series started with the basic question: can a four-dollar chip do real-time audio
DSP well? The answer after ten articles is: yes, with discipline.



What the RP2040 does well:


Two symmetric cores. Separating the real-time audio path from the
non-deterministic USB/MIDI path onto dedicated cores is the single most important
architectural decision in the series. It gives you determinism without an RTOS.
PIO and DMA. I2S audio output and input with no dedicated peripheral -
just PIO programs and DMA chains. Once set up, the CPU is completely free of per-sample
I/O work.
264 KB SRAM. Enough for multiple audio buffers, filter state, oscillator
tables (or pointers into flash), and a MIDI parser - with headroom.
Flash XIP. Large wavetables live in flash transparently. The 16 KB XIP cache
keeps sequential table reads fast.
Soft-float, not as bad as feared. A biquad filter costs about 1.9% CPU per
instance. You can run a full synthesiser voice at 3-4% CPU.



Where the RP2040 has limits:


No FPU. Filter coefficient recomputation per sample (sinf/cosf) is expensive.
The solution is caching coefficients and recomputing only on change - a software
discipline, not a hardware workaround.
No MPU stack protection. A stack overflow in the audio ISR is silent
corruption. Keep ISR code short and DSP state in static variables.
264 KB is tight for polyphony. Ten voices with separate filter state, envelope
state, and oscillator phase eat into SRAM quickly. For polyphonic synthesis beyond four
voices, budget SRAM carefully.
No hardware MIDI peripheral. UART MIDI requires a UART and a GPIO. USB MIDI
requires TinyUSB and careful attention to keeping tud_task() on core 0.
Neither is difficult, but both require plumbing.



The series ends here. You have a working instrument, a profiling methodology (SysTick),
an understanding of the dual-core architecture, and a set of reusable modules you can
adapt for your own projects. The RP2040 has more to offer - overclocking to 250 MHz
doubles the CPU budget, the PIO can do more complex I/O protocols, and the ADC enables
CV control for Eurorack integration. That is for a future series.


Summary


The complete synthesiser voice combines a wavetable oscillator (Part 5), a resonant
lowpass biquad filter (Parts 5 and this article), two ADSR envelopes (Part 6 pattern,
generalised here), USB MIDI input (Part 8), UART MIDI input (Part 7), I2S audio output
(Part 2), dual-core architecture (Part 4), and cross-core parameter passing (Part 9).
Total audio CPU usage at 44.1 kHz: approximately 3.5% in sustain, 14% during filter
envelope attack/decay without coefficient caching, less than 5% with caching. The Pico
has room for more.
